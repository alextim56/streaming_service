установка сборщика Webpack 
npm i -D webpack webpack-cli

поддержка изменений в реальном времени
npm i -D webpack-dev-server
"dev": "webpack serve --mode development" - разработка под webpack
"dev": "node --watch server.js" - разработка под node.js

установка библиотеки для работы с DOM
npm i redom

npm i ejs

установка поддержки scss
npm install sass-loader sass webpack --save-dev

npm install style-loader --save-dev
npm install css-loader --save-dev

установка typescript
npm install --save-dev typescript ts-loader

создание файла конфигурации tsconfig.json
npx tsc --init

npm install @types/node
